# part1: LVM

fdisk /dev/sdb				# Partition the second disk for LVM
lsblk					# Show disks and partitions to verify changes
pvcreate /dev/sdb1               	# Initialize the partition as a Physical Volume (PV)
pvs			                # List all physical volumes
vgcreate -s 16M vgdata /dev/sdb1	# Create a Volume Group (VG) with 16M PE
vgdisplay vgdata			# Display details of the Volume Group
lvcreate -n lv -l 100%VG vgdata  	# Create a Logical Volume (LV) using all VG space
lvs                    		        # List all logical volumes
blkid /dev/vgdata/lv  		        # Show UUID and filesystem info of the LV
mkdir /mnt/data         	        # Create a mount point for the LV
mount /dev/vgdata/lv /mnt/data          # Mount the LV at /mnt/data
mount -a                                # Mount all filesystems in /etc/fstab
mount | grep /mnt/data                  # Verify that /mnt/data is mounted
vim /etc/fstab                          # Edit fstab to mount the LV automatically at boot

#!/bin/bash — run with Bash.
sleep 600 & — spawn a 10-minute background process; PID=$! saves its PID.
Compute END_TIME so the script itself runs for 600 seconds.
while loop runs until END_TIME, checking the child each iteration.
If child exists ps -p $PID, try kill -9 $PID and report; otherwise print it’s gone.
sleep 10 prevents a tight CPU loop.
